\chapter{User Manual}
\section{Installation}
\begin{enumerate}
    \item Clone the repository: \texttt{git clone https://github.com/SAMEER-40/student-dropout.git}
    \item Install dependencies: \texttt{pip install -r requirements.txt}
    \item Start Backend: \texttt{uvicorn api.main:app --reload}
    \item Start Frontend: \texttt{cd frontend \&\& npm run dev}
\end{enumerate}

\section{Usage Guide}
\begin{itemize}
    \item Open browser to \texttt{http://localhost:5173}.
    \item Fill in the student details in the form.
    \item Toggle "Include Explanation" to see SHAP values.
    \item Click "Predict Outcome".
    \item View the result card (Red/Green) and interpretation chart.
\end{itemize}
